6   package markus.ginrummy.kayttoliittyma;
7   
8   import java.util.logging.Level;
9   import java.util.logging.Logger;
10  import javax.swing.JScrollPane;
11  import markus.ginrummy.logiikka.Client;
12  
13  /**
14   *
15   * @author Markus
16   */
17  public class FrameController implements Runnable{
18      
19      private Client client;
20  
21      public FrameController(Client client) {
22          this.client = client;
23      }
24      
25  
26      @Override
27      public void run() {
28          LoginScreen screen = new LoginScreen(client);
29          JScrollPane c = screen.getTheContainer();
30          ClientWriter writer = new ClientWriter(client, c, screen);
31          writer.start();
32          screen.setVisible(true);
33          synchronized (writer) {
34          try {
35              writer.wait();
36          } catch (InterruptedException ex) {
37              Logger.getLogger(FrameController.class.getName()).log(Level.SEVERE, null, ex);
38          }
39          }
40          screen.setVisible(false);
41              
42      }
43      
44  }
